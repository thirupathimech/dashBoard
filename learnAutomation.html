Navigating DOM Tree

Dynamic XPath

Handling Dynamic Elements 
// Dynamic ID handling
driver.findElement(By.xpath("//div[contains(@id,'dynamic_')]"));

// Dynamic class handling
driver.findElement(By.xpath("//span[contains(@class,'status-')]"));

// Using multiple conditions
driver.findElement(By.xpath("//input[@type='text' and contains(@name,'user')]"));

// Position-based for dynamic lists
driver.findElement(By.xpath("//ul[@class='menu']/li[2]"));

// Using text content
driver.findElement(By.xpath("//button[text()='Submit']"));
driver.findElement(By.xpath("//a[contains(text(),'Click here')]"));

Advanced Predicates

Complex Filtering 
//tr[position()>1]           // Skip header row
//div[position() mod 2 = 0]  // Even positioned divs

//td[text()='Active']
//span[normalize-space(text())='Success']

//input[@tabindex > 0]
//div[@data-priority >= 5]

//div[@class='item' and position()>1 and @data-status='active']

//input[not(@disabled)]
//div[not(contains(@class,'hidden'))]

Good Practices
  • Use relative XPath (//) instead of absolute
  • Prefer ID and Name attributes
  • Use specific tag names when possible
  • Minimize the use of complex predicates
Avoid These
  • Long absolute paths
  • Multiple contains() in one XPath
  • Complex text() matching
  • Deep descendant searches

Core Annotations

Essential TestNG Annotations 
@Test
// Marks a method as a test method

@BeforeMethod
// Runs before each test method

@AfterMethod
// Runs after each test method

@BeforeClass
// Runs once before the first test method in the class

@AfterClass
// Runs once after all test methods in the class

@BeforeSuite
// Runs once before all tests in the suite

@AfterSuite
// Runs once after all tests in the suite

@BeforeTest
// Runs before any test method in a test tag

@AfterTest
// Runs after all test methods in a test tag

@BeforeGroups
// Runs before the first test method in specified groups

@AfterGroups
// Runs after the last test method in specified groups

Complete Test Class Example

SeleniumTestExample.java 
package com.example.tests;

import org.testng.annotations.*;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.By;
import org.testng.Assert;

@Listeners({TestListener.class})
public class LoginTest {
    
    private WebDriver driver;
    
    @BeforeSuite
    public void setupSuite() {
        System.out.println("Setting up test suite");
        // Global setup
    }
    
    @BeforeClass
    public void setupClass() {
        System.out.println("Setting up test class");
        // Class-level setup
    }
    
    @BeforeMethod
    public void setUp() {
        System.setProperty("webdriver.chrome.driver", "path/to/chromedriver");
        driver = new ChromeDriver();
        driver.manage().window().maximize();
    }
    
    @Test(priority = 1, groups = {"smoke"}, 
          description = "Verify valid login functionality")
    public void testValidLogin() {
        driver.get("https://example.com/login");
        driver.findElement(By.xpath("//input[@id='username']")).sendKeys("validuser");
        driver.findElement(By.xpath("//input[@id='password']")).sendKeys("validpass");
        driver.findElement(By.xpath("//button[@type='submit']")).click();
        
        String welcomeMsg = driver.findElement(By.xpath("//div[@class='welcome']")).getText();
        Assert.assertEquals(welcomeMsg, "Welcome, validuser!");
    }
    
    @Test(priority = 2, groups = {"regression"}, 
          dependsOnMethods = {"testValidLogin"},
          description = "Verify invalid login shows error")
    public void testInvalidLogin() {
        driver.get("https://example.com/login");
        driver.findElement(By.xpath("//input[@id='username']")).sendKeys("invaliduser");
        driver.findElement(By.xpath("//input[@id='password']")).sendKeys("wrongpass");
        driver.findElement(By.xpath("//button[@type='submit']")).click();
        
        String errorMsg = driver.findElement(By.xpath("//div[@class='error']")).getText();
        Assert.assertTrue(errorMsg.contains("Invalid credentials"));
    }
    
    @Test(dataProvider = "loginData", groups = {"data-driven"})
    public void testMultipleLogins(String username, String password, boolean shouldSucceed) {
        driver.get("https://example.com/login");
        driver.findElement(By.xpath("//input[@id='username']")).sendKeys(username);
        driver.findElement(By.xpath("//input[@id='password']")).sendKeys(password);
        driver.findElement(By.xpath("//button[@type='submit']")).click();
        
        if (shouldSucceed) {
            Assert.assertTrue(driver.getCurrentUrl().contains("dashboard"));
        } else {
            Assert.assertTrue(driver.findElement(By.xpath("//div[@class='error']")).isDisplayed());
        }
    }
    
    @DataProvider(name = "loginData")
    public Object[][] getLoginData() {
        return new Object[][] {
            {"user1", "pass1", true},
            {"user2", "pass2", true},
            {"invalid", "wrong", false},
            {"", "", false}
        };
    }
    
    @Test(timeOut = 5000, invocationCount = 3,
          description = "Performance test for login")
    public void testLoginPerformance() {
        long startTime = System.currentTimeMillis();
        
        driver.get("https://example.com/login");
        driver.findElement(By.xpath("//input[@id='username']")).sendKeys("perfuser");
        driver.findElement(By.xpath("//input[@id='password']")).sendKeys("perfpass");
        driver.findElement(By.xpath("//button[@type='submit']")).click();
        
        long endTime = System.currentTimeMillis();
        Assert.assertTrue((endTime - startTime) < 3000, "Login took too long");
    }
    
    @Test(expectedExceptions = {org.openqa.selenium.NoSuchElementException.class},
          description = "Test that missing element throws exception")
    public void testMissingElement() {
        driver.get("https://example.com/login");
        driver.findElement(By.xpath("//input[@id='nonexistent']")); // Should throw exception
    }
    
    @AfterMethod
    public void tearDown() {
        if (driver != null) {
            driver.quit();
        }
    }
    
    @AfterClass
    public void tearDownClass() {
        System.out.println("Cleaning up test class");
    }
    
    @AfterSuite
    public void tearDownSuite() {
        System.out.println("Cleaning up test suite");
    }
}

Custom Test Listener

TestListener.java 
import org.testng.ITestListener;
import org.testng.ITestResult;

public class TestListener implements ITestListener {
    
    @Override
    public void onTestStart(ITestResult result) {
        System.out.println("Test Started: " + result.getName());
    }
    
    @Override
    public void onTestSuccess(ITestResult result) {
        System.out.println("Test Passed: " + result.getName());
    }
    
    @Override
    public void onTestFailure(ITestResult result) {
        System.out.println("Test Failed: " + result.getName());
        System.out.println("Failure Reason: " + result.getThrowable());
    }
    
    @Override
    public void onTestSkipped(ITestResult result) {
        System.out.println("Test Skipped: " + result.getName());
    }
}

Page Object Model Example

LoginPage.java 
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.support.FindBy;
import org.openqa.selenium.support.PageFactory;

public class LoginPage {
    private WebDriver driver;
    
    @FindBy(xpath = "//input[@id='username']")
    private WebElement usernameField;
    
    @FindBy(xpath = "//input[@id='password']")
    private WebElement passwordField;
    
    @FindBy(xpath = "//button[@type='submit']")
    private WebElement submitButton;
    
    @FindBy(xpath = "//div[@class='error']")
    private WebElement errorMessage;
    
    public LoginPage(WebDriver driver) {
        this.driver = driver;
        PageFactory.initElements(driver, this);
    }
    
    public void enterUsername(String username) {
        usernameField.clear();
        usernameField.sendKeys(username);
    }
    
    public void enterPassword(String password) {
        passwordField.clear();
        passwordField.sendKeys(password);
    }
    
    public void clickSubmit() {
        submitButton.click();
    }
    
    public String getErrorMessage() {
        return errorMessage.getText();
    }
    
    public boolean isErrorDisplayed() {
        return errorMessage.isDisplayed();
    }
}

testng.xml Configuration

Complete testng.xml Example 
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE suite SYSTEM "http://testng.org/testng-1.0.dtd">

<suite name="SeleniumTestSuite" parallel="methods" thread-count="3" verbose="2">
    
    <!-- Suite-level parameters -->
    <parameter name="baseUrl" value="https://example.com"/>
    <parameter name="browser" value="chrome"/>
    
    <!-- Listeners -->
    <listeners>
        <listener class-name="com.example.listeners.TestListener"/>
        <listener class-name="com.example.listeners.ReportListener"/>
    </listeners>
    
    <!-- Test groups -->
    <groups>
        <run>
            <include name="smoke"/>
            <include name="regression"/>
            <exclude name="slow"/>
        </run>
    </groups>
    
    <!-- Smoke Test Suite -->
    <test name="SmokeTests" preserve-order="true">
        <parameter name="timeout" value="30"/>
        
        <groups>
            <run>
                <include name="smoke"/>
            </run>
        </groups>
        
        <classes>
            <class name="com.example.tests.LoginTest"/>
            <class name="com.example.tests.HomePageTest"/>
        </classes>
    </test>
    
    <!-- Regression Test Suite -->
    <test name="RegressionTests" parallel="classes" thread-count="2">
        
        <groups>
            <run>
                <include name="regression"/>
            </run>
        </groups>
        
        <packages>
            <package name="com.example.tests.*"/>
        </packages>
    </test>
    
    <!-- Method-specific test -->
    <test name="SpecificMethodTest">
        <classes>
            <class name="com.example.tests.LoginTest">
                <methods>
                    <include name="testValidLogin"/>
                    <include name="testInvalidLogin"/>
                    <exclude name="testLoginPerformance"/>
                </methods>
            </class>
        </classes>
    </test>
    
    <!-- Data Provider Test -->
    <test name="DataDrivenTests">
        <parameter name="dataFile" value="testdata.xlsx"/>
        
        <classes>
            <class name="com.example.tests.DataDrivenTest"/>
        </classes>
    </test>
    
</suite>

Command Line Execution

Various Execution Options 
# Run entire suite
java -cp "classes:lib/*" org.testng.TestNG testng.xml

# Run specific groups
java -cp "classes:lib/*" org.testng.TestNG -groups smoke testng.xml

# Run with parameters
java -cp "classes:lib/*" org.testng.TestNG -Dbrowser=firefox testng.xml

# Generate reports
java -cp "classes:lib/*" org.testng.TestNG -d results testng.xml

# Parallel execution
java -cp "classes:lib/*" org.testng.TestNG -parallel methods -threadcount 5 testng.xml

# With Maven
mvn test -DsuiteXmlFile=smoke-tests.xml
mvn test -Dgroups=regression
mvn test -Dtest=LoginTest#testValidLogin
